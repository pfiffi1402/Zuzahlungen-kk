    <script></div>

        const allowances = {</div>

            "2025": { single: 6500, married: 13000, child: 3700 },</div>

            "2026": { single: 6650, married: 13300, child: 3750 },</div>

            // Weitere zukünftige Jahre können hier ergänzt werden.</div>

        };</div>


</div>

        function populateYearOptions() {</div>

            const yearSelect = document.getElementById("year");</div>

            const currentYear = new Date().getFullYear();</div>

            const currentMonth = new Date().getMonth() + 1; // Monat von 0-11</div>


</div>

            // Startjahr ist immer 2025 oder später</div>

            const baseYear = 2025;</div>


</div>

            // Füge nur das Basisjahr hinzu</div>

            const option1 = document.createElement("option");</div>

            option1.value = baseYear.toString();</div>

            option1.textContent = baseYear;</div>

            yearSelect.appendChild(option1);</div>


</div>

            // Füge das nächste Jahr hinzu, wenn der Monat September oder später ist</div>

            if (currentYear >= baseYear && currentMonth >= 9) {</div>

                const option2 = document.createElement("option");</div>

                option2.value = (baseYear + 1).toString();</div>

                option2.textContent = baseYear + 1;</div>

                yearSelect.appendChild(option2);</div>

            }</div>

        }</div>


</div>

        function calculateZuzahlungen() {</div>

            const income = parseFloat(document.getElementById("income").value);</div>

            const familyStatus = document.getElementById("familyStatus").value;</div>

            const children = parseInt(document.getElementById("children").value) || 0; // Standardwert 0</div>

            const payments = parseFloat(document.getElementById("payments").value) || 0; // Standardwert 0</div>

            const year = document.getElementById("year").value;</div>


</div>

            // Holen der Freibeträge für das gewählte Jahr</div>

            const yearData = allowances[year];</div>

            if (!yearData) {</div>

                alert("Freibeträge für das gewählte Jahr sind nicht verfügbar.");</div>

                return;</div>

            }</div>


</div>

            let allowance = familyStatus === "verheiratet" ? yearData.married : yearData.single;</div>

            allowance += children * yearData.child;</div>


</div>

            // Berechnung der Belastungsgrenze</div>

            const annualIncome = income * 12;</div>

            const taxableIncome = Math.max(annualIncome - allowance, 0);</div>

            const burdenLimit = taxableIncome * 0.02;</div>


</div>

            // Verbleibende Zuzahlungen</div>

            const remainingPayments = Math.max(burdenLimit - payments, 0);</div>


</div>

            // Ergebnisse anzeigen</div>

            const resultDiv = document.getElementById("zuzahlungResult");</div>

            resultDiv.style.display = "block";</div>

            resultDiv.innerHTML = `</div>

                <p><strong>Jahr: ${year}</strong></p></div>

                <p>Ihre individuelle Belastungsgrenze: <strong>${burdenLimit.toFixed(2)} Euro</strong></p></div>

                ${remainingPayments === 0 </div>

                    ? "<p>Sie haben die Belastungsgrenze erreicht. Keine weiteren Zuzahlungen erforderlich.</p>"</div>

                    : `<p>Verbleibende Zuzahlungen bis zur Belastungsgrenze: <strong>${remainingPayments.toFixed(2)} Euro</strong></p>`}</div>

            `;</div>

        }</div>


</div>

        // Fülle die Jahr-Auswahl, wenn die Seite geladen wird</div>

        document.addEventListener("DOMContentLoaded", populateYearOptions);</div>

    </script></div>
